% The following commands make the chapter titles nice. The -4.35pc a few lines
% below is a hack to make the spacings between starred and unstarred \chapter
% titles match, as titlesec treats those cases differently.
\titlespacing*{\chapter}{0pt}{0pt}{0pt}

\titleformat{\chapter}[display]
{\normalfont\filcenter}
{\Huge \bfseries \vspace*{-4.35pc} \thechapter}
{.5em}
{
\titlerule
\vspace{0.5pc}%
\Large \libertineDisplay}
[\vspace{4cm}]

\titlespacing*{\section}{0pt}{5.5ex plus 1ex minus .2ex}{2.3ex plus .2ex}

\titleformat{\section}[block]
{\Large \bfseries \filcenter}
{\thesection}
{1em}
{}
